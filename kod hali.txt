    function exportData() {
        const dataStr = JSON.stringify(SITE_DATA, null, 2);
        const blob = new Blob([dataStr], { type: 'application/json' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = `nihatkeles_yedek_${new Date().toISOString().slice(0,10)}.json`;
        a.click();
    }
    
    function importData() {
        const input = document.createElement('input');
        input.type = 'file';
        input.accept = '.json';
        input.onchange = function(e) {
            const file = e.target.files[0];
            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const data = JSON.parse(e.target.result);
                    SITE_DATA = data;
                    saveData();
                    location.reload();
                } catch(err) {
                    alert('❌ Hatalı dosya!');
                }
            };
            reader.readAsText(file);
        };
        input.click();
    }

    // ========== MAĞAZA FONKSİYONLARI ==========
    function renderShop() {
        const container = document.getElementById('shop-container');
        if (!container) return;
        
        const productsHtml = SITE_DATA.products.map(p => `
            <div class="product-card">
                <h3 style="color:#ffd966;">${p.title}</h3>
                <p style="color:#e6d5b8;">${p.description}</p>
                <div class="product-price">${p.price} TL</div>
                <button class="buy-button" onclick="openPaymentModal('${p.id}')">🛒 Satın Al</button>
            </div>
        `).join('');
        
        container.innerHTML = productsHtml;
    }
    
    function openPaymentModal(productId) {
        const product = SITE_DATA.products.find(p => p.id === productId);
        if (!product) return;
        
        document.getElementById('modalProductTitle').innerText = `🛒 ${product.title}`;
        document.getElementById('modalProductDetails').innerHTML = `
            <p style="color:#e6d5b8;">${product.description}</p>
            <p style="color:#ffd966; font-size:1.3rem;">${product.price} TL</p>
        `;
        document.getElementById('paymentModal').dataset.productId = productId;
        document.getElementById('paymentModal').classList.add('show');
    }
    
    function closePaymentModal() {
        document.getElementById('paymentModal').classList.remove('show');
    }
    
    function processPayment() {
        const name = document.getElementById('buyerName').value;
        const email = document.getElementById('buyerEmail').value;
        const productId = document.getElementById('paymentModal').dataset.productId;
        const product = SITE_DATA.products.find(p => p.id === productId);
        
        if (!name || !email) { alert('Ad ve e-posta gerekli!'); return; }
        
        const order = { id: Date.now(), productId, productTitle: product.title, price: product.price, customerName: name, customerEmail: email, date: new Date().toISOString(), status: 'pending' };
        SITE_DATA.orders.push(order);
        
        if (!SITE_DATA.subscribers.includes(email)) SITE_DATA.subscribers.push(email);
        saveData();
        
        alert(`✅ Sipariş alındı! PDF ${email} adresine gönderildi.`);
        closePaymentModal();
        document.getElementById('buyerName').value = '';
        document.getElementById('buyerEmail').value = '';
    }

    function createConfetti() {
        const container = document.getElementById('confetti-container');
        container.innerHTML = '';
        for (let i = 0; i < 25; i++) {
            const confetti = document.createElement('div');
            confetti.className = 'confetti';
            confetti.style.left = Math.random() * 100 + '%';
            confetti.style.animationDuration = Math.random() * 3 + 4 + 's';
            confetti.style.animationDelay = Math.random() * 5 + 's';
            confetti.style.background = `hsl(${Math.random() * 360}, 80%, 60%)`;
            container.appendChild(confetti);
        }
    }

    function loadSavedData() {

            const saved = localStorage.getItem('nihatkeles_site');
            if (saved) SITE_DATA = { ...SITE_DATA, ...JSON.parse(saved) };
        } catch(e) {}

    function refreshMenu() {
        const menuHtml = SITE_DATA.menu.map(item => 
            `<li><button onclick="showPage('${item.id}')" class="nav-btn" id="btn-${item.id}">${item.title}</button></li>`
        ).join('');
        document.getElementById('main-menu').innerHTML = menuHtml;
    }
    
    function showPage(pageId) {
        SITE_DATA.currentPage = pageId;
        let content = SITE_DATA.pages[pageId] || '<h2>Sayfa Bulunamadı</h2>';
        
        if (pageId === 'drama' && SITE_DATA.videos.length > 0) {
            const videoHtml = SITE_DATA.videos.map(v => `
                <div class="video-card">
                    <h3 style="color:#ffd966;">${v.title}</h3>
                    <div class="video-container"><iframe src="${v.url}" allowfullscreen></iframe></div>
                </div>
            `).join('');
            content += videoHtml;
        }
        
        document.getElementById('main-content').innerHTML = content;
        
        if (pageId === 'magaza') renderShop();
        
        document.querySelectorAll('.nav-btn').forEach(btn => btn.classList.remove('active'));
        const activeBtn = document.getElementById(`btn-${pageId}`);
        if (activeBtn) activeBtn.classList.add('active');
        
        setTimeout(initAccordions, 100);
    }
    
    function initAccordions() {
        document.querySelectorAll('.accordion-header').forEach(header => {
            header.replaceWith(header.cloneNode(true));
        });
        document.querySelectorAll('.accordion-header').forEach(header => {
            header.addEventListener('click', function() {
                this.classList.toggle('active');
                const content = this.nextElementSibling;
                if (content.style.maxHeight) {
                    content.style.maxHeight = null;
                } else {
                    document.querySelectorAll('.accordion-content').forEach(c => c.style.maxHeight = null);
                    document.querySelectorAll('.accordion-header').forEach(h => h.classList.remove('active'));
                    this.classList.add('active');
                    content.style.maxHeight = content.scrollHeight + 'px';
                }
            });
        });
    }
